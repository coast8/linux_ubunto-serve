# 1 - Install o Server MySQL	
	sudo apt-get install mysql-server

## Starting MYSQL
	sudo /etc/init.d/mysql start

## Stop MySQL
	sudo /etc/init.d/mysql stop

## To restart MySQL
	sudo /etc/init.d/mysql restart

## To check the status of MySQL
	sudo /etc/init.d/mysql status
